111  void subscribe_to(source_value_type st)
112  {
113  auto state = this->shared_from_this();
114 
115  auto selectedCollection = on_exception(
116  [&](){return state->selectCollection(st);},
117  state->out);
118  if (selectedCollection.empty()) {
119  return;
120  }
121 
122  collectionLifetime = composite_subscription();
123 
124  // when the out observer is unsubscribed all the
125  // inner subscriptions are unsubscribed as well
126  auto innercstoken = state->out.add(collectionLifetime);
127 
128  collectionLifetime.add(make_subscription([state, innercstoken](){
129  state->out.remove(innercstoken);
130  }));
131 
132  auto selectedSource = on_exception(
133  [&](){return state->coordinator.in(selectedCollection.get());},
134  state->out);
135  if (selectedSource.empty()) {
136  return;
137  }
138 
139  // this subscribe does not share the source subscription
140  // so that when it is unsubscribed the source will continue
141  auto sinkInner = make_subscriber<collection_value_type>(
142  state->out,
143  collectionLifetime,
144  // on_next
145  [state, st](collection_value_type ct) {
146  auto selectedResult = state->selectResult(st, std::move(ct));
147  state->out.on_next(std::move(selectedResult));
148  },
149  // on_error
150  [state](std::exception_ptr e) {
151  state->out.on_error(e);
152  },
153  //on_completed
154  [state](){
155  if (!state->selectedCollections.empty()) {
156  auto value = state->selectedCollections.front();
157  state->selectedCollections.pop_front();
158  state->collectionLifetime.unsubscribe();
159  state->subscribe_to(value);
160  } else if (!state->sourceLifetime.is_subscribed()) {
161  state->out.on_completed();
162  }
163  }
164  );
165  auto selectedSinkInner = on_exception(
166  [&](){return state->coordinator.out(sinkInner);},
167  state->out);
168  if (selectedSinkInner.empty()) {
169  return;
170  }
171  selectedSource->subscribe(std::move(selectedSinkInner.get()));
172  }
173  composite_subscription sourceLifetime;
174  composite_subscription collectionLifetime;
175  std::deque<source_value_type> selectedCollections;
176  coordinator_type coordinator;
177  output_type out;
178  };

180  auto coordinator = initial.coordination.create_coordinator(scbr.get_subscription());
181

192  [&](){return state->coordinator.in(state->source);},
193  state->out);

196  }
197

199  // so that when it is unsubscribed the observer can be called
200  // until the inner subscriptions have finished
201  auto sink = make_subscriber<source_value_type>(
202  state->out,
203  state->sourceLifetime,
204  // on_next

207  state->selectedCollections.push_back(st);
208  } else if (state->selectedCollections.empty()) {

211  },

215  },

219  state->out.on_completed();
